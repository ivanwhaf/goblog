<script>
function get_article_preview()
{
var pathname=window.location.pathname;
var pn=pathname.split("/");
var page=pn[1];

if(page=="")
{
page=1;
}

$.ajax(
{
method:'GET',
url:'/get_article_preview?page='+page,
dataType:'json',
success:function(data)
{
var total_count=parseInt(data.total_count);
var visitor_count=data.visitor_count;
var data=data.data;
$("#vc").append(visitor_count);
for(var i=0;i<data.length;i++)
{
id=data[i].id;
title=data[i].title;
subtitle=data[i].subtitle;
tag=data[i].tag;
content=data[i].content;
date=data[i].date;
$("#essay").append(
'<div class="post-preview">\n'+
' <a href="article/'+tag+'/'+id+'">\n'+
' <h2 class="post-title">['+tag+'] '+title+'</h2>\n'+
' <h3 class="post-subtitle">'+subtitle+'</h3>\n'+
' <div class="post-content-preview">'+content+'</div>\n'+
' </a>\n'+
' <p class="post-meta" style="margin: 10px 0;">Posted by Ivan on '+date+'</p>\n'+
' <div class="tags"><a href="/tags#'+tag+'">'+tag+'</a>\n</div>\n'+
'</div>'+
'<hr>');
}
$("#essay").append('<ul class="pager" id="pa"></ul><br>');
page=parseInt(page);
if(page!=1)
{
$("#pa").append('<li class="previous">'+
'<a href="/'+(page-1).toString()+'">&larr; Newer Posts</a>'+
'</li>');
}
if(page*10<total_count)
{
$("#pa").append('<li class="next">'+
'<a href="/'+(page+1).toString()+'">Older Posts →</a>'+
'</li>');
}
}
}
);
}
$(document).ready(get_article_preview());
</script>
